function gerarListaLateral() { const list = document.getElementById('lista-imoveis'); if (!list) return; list.innerHTML = ""; DADOS_PLANILHA.forEach(item => { if (item.tipo === 'N') { const div = document.createElement('div'); div.className = "separador-complexo"; // O estilo já está no seu CSS div.style = "background:#333; color:white; padding:8px; font-size:0.75rem; text-align:center; margin:15px 0 5px 0; border-radius:4px; font-weight:bold; text-transform:uppercase;"; div.innerText = item.nome; list.appendChild(div); } else { const btn = document.createElement('button'); btn.className = `btRes`; // IMPORTANTE: O ID deve ser gerado limpando caracteres especiais btn.id = `btn-esq-${item.nome.replace(/[^a-zA-Z0-9]/g, '-')}`; btn.innerHTML = `${item.nome} ${obterHtmlEstoque(item.estoque, item.tipo)}`; btn.onclick = () => comandoSelecao(item.id_path, item.cidade, item); list.appendChild(btn); } }); }
